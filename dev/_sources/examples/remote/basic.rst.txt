    from metpy.remote import GOESArchive, NEXRADLevel2Archive, NEXRADLevel3Archive

    prod = NEXRADLevel2Archive().get_product('KTLX', datetime(2013, 5, 22, 21, 53))

    # NEXRAD Level 3
